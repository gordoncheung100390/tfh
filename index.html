App.studio = { peer: null, localStream: null, connectionActive: false, // Prevents "Flood" errors in console init: function() { // 1. Setup Identity const gId = "OMNI-" + App.state.roomID + "-GUIDE"; const myId = App.state.user.role === 'guide' ? gId : undefined; // 2. Initialize Peer with HTTPS settings for GitHub this.peer = new Peer(myId, { secure: true, port: 443, config: { 'iceServers': [{ urls: 'stun:stun.l.google.com:19302' }] } }); // 3. Set up listeners based on Role if(App.state.user.role === 'guide') { // Guide waits for calls this.peer.on('call', incomingCall => { if(this.localStream) { incomingCall.answer(this.localStream); this.setupStream(incomingCall); } }); } else { // Tourist looks for Guide every 5 seconds (Only if NOT connected) setInterval(() => { if(!this.connectionActive && this.peer && this.peer.open) { this.connectToGuide(); } }, 5000); } }, toggleCam: function() { const v = document.getElementById('remote-video'); if(!this.localStream) { navigator.mediaDevices.getUserMedia({video:true, audio:true}).then(s => { this.localStream = s; v.srcObject = s; v.muted = true; // Mute local preview to prevent echo v.play(); // Update UI state document.getElementById('cam-btn').innerText = "Cam ON"; document.getElementById('cam-btn').style.background = "red"; document.getElementById('live-indicator').innerText = "LIVE"; document.getElementById('live-indicator').style.background = "red"; }).catch(err => alert("Camera Access Denied. Ensure HTTPS is active.")); } else { // Cleanup: Stop all camera tracks this.localStream.getTracks().forEach(t => t.stop()); this.localStream = null; v.srcObject = null; document.getElementById('cam-btn').innerText = "Cam OFF"; document.getElementById('cam-btn').style.background = "#333"; document.getElementById('live-indicator').innerText = "OFFLINE"; } }, connectToGuide: function() { const targetId = "OMNI-" + App.state.roomID + "-GUIDE"; // Call with an empty stream to initiate the handshake const call = this.peer.call(targetId, new MediaStream()); if(call) this.setupStream(call); }, setupStream: function(call) { call.on('stream', remoteStream => { this.connectionActive = true; const v = document.getElementById('remote-video'); v.srcObject = remoteStream; v.muted = false; // Interaction Guard: Browser requires a click before playing audio v.play().catch(e => console.log("Click the page to enable video.")); document.getElementById('live-indicator').innerText = "GUIDE LIVE"; document.getElementById('live-indicator').style.background = "green"; }); // Reset the flag if the connection drops so we can reconnect call.on('close', () => { this.connectionActive = false; }); call.on('error', () => { this.connectionActive = false; }); } };
